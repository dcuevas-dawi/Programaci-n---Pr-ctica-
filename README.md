decimalBinario(int n){
	while(n != 0){
		print(n % 2)
		n = n / 2
	}
}
~~~

### Ejercicio 2

~~~
binarioDecimal(long n){
	int i, n2, suma = 0
	while(n != 0){
		n2 = n % 10
		n = n / 10
		suma = suma + (n2*2^i)
		i++
	}
	print(suma)
}
~~~
